\section{Einleitung}

\begin{frame}
  
  RSS: Really Simple Syndication
  \vspace{0.5cm}
  \begin{itemize}
  \item Format-Standard zur Verbreitung von Informationen im Internet
  \item Dateiformat ist XML-basiert
  \item RSS-Feed ist einem Channel zugeordnet
  \item Verbreitung mittels Pull
  \item Beispiele: Nachrichten-Schlagzeilen, Blogs, Podcasting (anbieten von Medien-Dateien, z.B. mp3)
  \end{itemize}
  
\end{frame}

\subsection*{Motivation}
\begin{frame}
  
  \frametitle{Anwendersicht: Was man gerne hätte ...}
  Aus Sicht eines Klienten:
  \vspace{1cm}
  \begin{itemize}
  \item Automatische Benachrichtigung über Feed-Aktualisierungen
  \item Aktualität der Feeds
  \item Definition von komplexeren Filtern: zur \dots
    \begin{list}{}{}
    \item \dots individuellen Vorselektion der Feeds
    \item \dots anbieterübergreifenden Auswahl von Feeds
    \end{list}
  \end{itemize}
  
\end{frame}

\begin{frame}
  
  Aus Sicht eines Server-Betreibers:
  \vspace{1cm}
  \begin{itemize}
  \item System sollte nicht überlastet werden
  \item Es sollten möglichst keine Änderungen am Softwaresystem erforderlich sein
  \end{itemize}
  
\end{frame}

\begin{frame}
  
  \frametitle{Stand der Dinge}
  \begin{itemize}
  \item RSS: Kein push-basiertes Pub/Sub $\rightarrow$ Client/Server
  \item Feeds müssen von Anwendersoftware heruntergeladen werden (Polling)
  \item Vordefinierte Channel $\rightarrow$ thematische Zuordnung auf RSS-Server-Seite
  \item Definition von Filtern nur auf Anwenderseite möglich
  \end{itemize}
  
  \pause
  Probleme:
  \begin{itemize}
  \item Polling:
    \begin{list}{$\longrightarrow$}{}
    \item Großer Datenverkehr im Netz
    \item Hohe Server-seitige Last
    \end{list}
  \item Filterdefinition nur beim Anwender:
    \begin{list}{$\longrightarrow$}{}
    \item Alle Feeds müssen heruntergeladen werden
    \end{list}
  \end{itemize}
  \nocite{Hicks:2004:RSSBandwith}
  
\end{frame}

\begin{frame}
  
  \frametitle{Ziele}
  \begin{itemize}
  \item optimale Ausnutzung der RSS-Server
  \item Reduzierung des Datenverkehrs aufgrund von Polling
  \item Erhöhung des Aktualitätsgrades der Feeds auf Nutzerseite
  \item Vorselektion von Informationen durch komplexere Filter soll ermöglicht werden
  \end{itemize}
  
\end{frame}

\section{Verwandte Arbeiten}
\subsection*{FeedTree}

\begin{frame}
  \frametitle{FeedTree \cite{SandlerEtAl:2005:FeedTree}}
  \begin{itemize}
  \item Basiert auf Scribe \cite{SCRIBE}
  \item Signierte RSS-Dokumente werden per Multicast an alle Subscriber gesendet, die dieses abonniert haben
  \item Abonnement erfolgt durch Beitritt zu einer Gruppe
  \item Thema einer Gruppe entspricht der ID des angebotenen RSS-Feeds (URL)
  \end{itemize}
\end{frame}

\begin{frame}
  Vorteile:
  \begin{itemize}
  \item Zeitnahe Benachrichtgung über Feed-Aktualisierungen möglich
  \item Bandbreiten werden gespart
  \item Authentizitätsprüfung durch Signierung
  \item Wiederbeschaffung von Feeds durch Zwischenspeicherung in Caches
  \end{itemize}
  \pause
  \vspace{0.5cm}
  Nachteile:
  \begin{itemize}
  \item Zusätzliche Software auf Anbieterseite erforderlich
  \item keine erweiterten Filtertechniken möglich
  \item Nur ein Publisher vorhanden $\longrightarrow$ wirkt sich nachteilig auf Aktualitätsgrad aus
  \end{itemize}
\end{frame}  

\subsection*{CMS-ToPSS}

\begin{frame}
  \frametitle{CMS-ToPSS \cite{PetrovicEtAl:2005:CMS-ToPSS}}
  \begin{itemize}
  \item Erweiterung zu Content-Management-Systemen
  \item Nutzt Notifikationssystem G-ToPSS: inhaltsbasierte Filterung durch GQL
  \end{itemize}
\end{frame}

\begin{frame}
  Vorteile:
  \begin{itemize}
  \item Flexible und mächtige Filterdefinitionen sind möglich
  \end{itemize}
  \pause
  \vspace{0.5cm}
  Nachteile:
  \begin{itemize}
  \item Neue Software auf Anbieterseite
  \item Nur ein Publisher vorgesehen
  \item Dateiformat muss RDF-Syntax entsprechen
  \end{itemize}
\end{frame}

\section{Umsetzung}
\subsection*{Pub/Sub-RSS}

\begin{frame}
  \frametitle{Informationsverbreitung mittels RSS und Publish/Subscribe}
  Innerhalb eines Overlay-Netzwerkes wird für die Verteilung der Feeds gesorgt.\\
  \vspace{0.5cm}
  Als Grundlage kann eine schon existierende Pub/Sub-Middleware dienen (z.B. REBECA\cite{MuFiBu:2001:ArchFrameECommApp}).
\end{frame}

\begin{frame}
  \frametitle{Ansatz}
  \begin{itemize}
  \item Rolle der RSS-Server verändert sich nicht
  \item Nutzer holen Feed und übermitteln ihn in der Rolle des Publisher an das Notifikationssystem
  \item Notifikationssystem verteilt Feeds an Nutzer bzw. Subscriber
  \end{itemize}
\end{frame}

\begin{frame}
  \includegraphics[bb=-100 0 200 250,scale=0.6]{RSSPubSub}
\end{frame}

\begin{frame}
  \frametitle{Ansatz: Ein Publisher}
  \begin{itemize}
  \item Ein Publisher würde ausreichen, um einen Feed herunterzuladen
  \item Alle weiteren Nutzer  bzw. Subscriber würden den Feed über das Pub/Sub-System erhalten
  \end{itemize}
  
  \pause
  Vorteil:\\
  Wenige Anfragen an den RSS-Server
  \vspace{0.5cm}
  
  \pause
  Problem:\\
  Aktualität der Feeds!\\
  $\longrightarrow$ lange Übertragungszeiten in Abhängigkeit von Netzstruktur\\
  u. U. aufwändiger Auswahlprozess\\
  Single Point of Failure
\end{frame}

\begin{frame}
  \frametitle{Ansatz: Mehrere Publisher}
  Frage:\\
  Wie können sich die Publisher darüber abstimmen, wer als nächstes den aktuellen Feed abholt?
\end{frame}

\begin{frame}
  \frametitle{Gesichtspunkte für die Entwicklung eines Algorithmus}
  $\rightarrow$ geforderte Ziele
  \begin{itemize}
  \item Polling durch mehrere bzw. wechselnde Klienten
  \item Ausfall von Klienten im Overlay-Netzwerk soll Informationsverteilung nicht blockieren
  \end{itemize}
  \vspace{0.5cm}
  $\rightarrow$ angestrebte Ziele
  \begin{itemize}
  \item Anfragen an den RSS-Server sollten nicht gleichzeitig für alle Klienten geschehen 
  \item Netzbelastung durch Abstimmungsnachrichten sollte gering gehalten werden
  \end{itemize}
\end{frame}

\subsection*{Koordinierung der Publisher}

\begin{frame}
  \frametitle{Koordinierung der Publisher}
  \begin{itemize}
  \item Jeder Publisher $i$ wählt zufälligen Zeitpunkt $ttr_i$ innerhalb eines zeitlichen Intervalls, um nächsten Feed herunterzuladen
  \end{itemize}
  \vspace{0.5cm}
  \includegraphics[bb=28 656 356 725,scale=1]{ti_picture}  
\end{frame}

\begin{frame}
  \frametitle{Veranschaulichung}
  $ttr$s der Publisher P1 $\dots$ P5\\
  \vspace{0.5cm}
  \includegraphics[bb=-40 0 200 150,scale=0.5]{Platzierungen}
\end{frame}

\begin{frame}
  \frametitle{Bestimmung von nextBuild}
  $\rightarrow$ Bestimmung über time-to-live ($ttl$):\\
  \vspace{0.2cm}
  \hspace{0.5cm}$nextBuild:=t_0+\varDelta ttl$
  \begin{displaymath}
    \varDelta ttl:=\left\{\begin{array}{r@{\quad:\quad}l}
        0 & falls\quad (t_0-lastBuildDate)>ttl \\ttl-(t_0-lastBuildDate) & sonst
      \end{array}\right.
  \end{displaymath}
  \begin{list}{}{}
  \item $lastBuildDate$ ist der Zeitpunkt des am weitesten zurückliegenden neuen Ereignisses innerhalb des Feeds
  \end{list}
  \vspace{0.5cm}
  $\rightarrow$ Bestimmung über RSS-Parameter $cloud$:\\
  \vspace{0.2cm}
  \hspace{0.5cm}$nextBuild:=t_0$
\end{frame}

\begin{frame}
  \includegraphics[bb=-40 0 200 150,scale=0.5]{BuildDates}
  \\
  \vspace{0.5cm}
  $ttl=[lastBuildDate,nextBuild]$\\
  $\varDelta ttl=[t_0,nextBuild]$
\end{frame}

\begin{frame}
  \frametitle{Bestimmung von time-to-live}
  \begin{itemize}
  \item Bereitstellung durch den Informationsanbieter\\(RSS-Parameter $ttl$)
  \item heuristische Bestimmung
  \end{itemize}
\end{frame}

\subsection*{Angestrebte Dienstgüte: bevorzugte Polling-Periode}

\begin{frame}
  \frametitle{Angestrebte Dienstgüte: bevorzugte Polling-Periode}
  Subscriber möchte spätestens nach einer von ihm festgesetzten Zeitspanne $ppp$ über Feed-Aktualisierungen unterrichtet werden\\
  \vspace{0.5cm}
  $\longrightarrow$ Polling muss mit der Frequenz $\frac{1}{ppp}$ geschehen
\end{frame}
  
\begin{frame}
  \frametitle{Problem der zu kleinen Polling-Perioden}
  Zu kleine Polling-Perioden können Aktualitätsgrad negativ beeinflussen (Serverüberlastung!)\\
  \includegraphics[bb=30 0 200 250,scale=0.6]{Zusammenhang_Pollingrate_Aktualitaetsgrad}
\end{frame}

\begin{frame}
  \frametitle{Dienstgüte}
  $\rightarrow$ Verzögerungsgrad\\
  \vspace{0.2cm}
  \begin{math}
    \Gamma_V:=\begin{cases}
      0& \text{wenn $(\varDelta pubDate-ppp) < 0$},\\
      \frac{100\cdot(\varDelta pubDate-ppp)}{ppp} & \text{sonst}.
    \end{cases}
  \end{math}
  \\
  \vspace{0.5cm}
  $\rightarrow$ Aktualisierungsgrad\\
  \vspace{0.2cm}
  \begin{math}
    \Gamma_A:=\begin{cases}
      100& \text{wenn $(\varDelta pubDate-ppp) < 0$},\\
      \frac{100\cdot ppp}{\varDelta pubDate}& \text{sonst}.
    \end{cases}
  \end{math}
  \\
  \vspace{0.5cm}
  $\varDelta pubDate$ = Zeitspanne von $t_0$ bis zum am weitesten zurückliegenden neuen
  Ereignis innerhalb des letzten RSS-Feeds  
\end{frame}

\begin{frame}
  \frametitle{Anpassung der Polling-Perioden durch Staukontrolle}
  wichtige Parameter:
  \begin{itemize}
  \item $ppp$: bevorzugte Polling-Periode
  \item $cpp$: aktuelle Polling-Periode
  \item $mpp$: maximale Polling-Periode
  \item $ttr$: time-to-refresh
  \item $rto$: retransmission-timeout-intervall -- Zeit bis zu einer erneuten Aussendung
  \item $rtt$: roundtrip-time -- Antwortzeit des Servers
  \item $artt$: angepasste roundtrip-time
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Timer}
  \begin{list}{$\rightarrow$}{}
  \item RT (Retransmission-Timer): bei erfolglosem Verbindungsversuch\\
    \begin{itemize}
    \item wird bestimmt durch $rto$
    \item nach jedem Ablauf wird $rto$ verdoppelt
    \item initial: $rto:=cpp$
    \end{itemize}
  \end{list}
  \vspace{0.5cm}
  \begin{list}{$\rightarrow$}{}
  \item RQT (Request-Timer): nach erfolgreichem Transfer eines Feeds\\
    \begin{itemize}
    \item wird bestimmt durch $ttr$ bzw. $\varDelta ttr:=cpp$
    \item $artt:=2^i-1\cdot rtt$, bei $i$ Verbindungsversuchen
    \item $cpp:=min\{mpp,max\{artt,ppp\}\}$
    \end{itemize}
  \end{list}
\end{frame}

\begin{frame}
  \frametitle{Seiteneffekt: \glqq Aussperren\grqq{} von Klienten}
  Klienten, deren Polling-Perioden sehr niedrig sind, haben fast ausschließlichen Zugriff
  auf den Server, während sich die Polling-Perioden der übrigen Klienten stetig vergrößern.\\
  \vspace{1cm}
  Lösung:
  \begin{itemize}
  \item Ausbalancierung der Polling-Perioden
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Ausbalancierung der Polling-Perioden}
  \begin{itemize}
  \item ermittelter $artt$ wird zusammen mit dem Feed an Notifikationssystem weitergeleitet und verbreitet
  \item bei Erhalt eines Feeds über das Notifikationssystem berechnte sich der neue $artt$ aus:
  \end{itemize}
  \hspace{1cm}
  $artt:=\frac{artt+feed.artt}{2}$\\
  \vspace{0.5cm}
  Vorteile:
  \begin{itemize}
  \item alle Klienten profitieren von ermittelten kleinen $rtt$s
  \item zu schneller Ansturm auf Server wird verhindert
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Phänomen: Churn}
  Dynamisches Zu-und Abwandern von Klienten kann die Server-Belastung steigern:
  \begin{itemize}
  \item bei hoher Churn-Rate durch Setzen von $cpp:=ppp$ bei Eintritt in das System
  \end{itemize}
  \vspace{0.5cm}
  Churn-Kompensation:
  \begin{itemize}
  \item bei Eintritt eines Klienten in das System übermittelt Notifikationssystem Feed und $feed.artt$
  \item für die Entscheidung, ob $feed.artt$ zuverlässig ist, reicht lokales Wissen (Anzahl angemeldeter Subscriber beim lokalen Broker)
  \end{itemize}
\end{frame}

\section{Zusammenfassung}

\begin{frame}
  \frametitle{Zusammenfassung}
  \begin{itemize}
  \item Konzept eines Systems zur Verteilung von RSS-Feeds basierend auf dem Kommunikationsmodell von verteiltem Pub/Sub
  \item bisheriger Ansatz bleibt bestehen: Klienten laden Feed direkt von der Webseite des Anbieters
  \item Klienten speisen Feeds in ein Notifikationssystem ein
  \end{itemize}
\end{frame}

  \item Verfahren zur Adaption der Polling-Perioden verschiedener Klienten an die Leistungsfähigkeit eines Servers
  \end{itemize}
\end{frame}
