\section{Einleitung}

\begin{frame}

    \item Verteilung von Informationen über das Internet gewinnt immer mehr an Bedeutung.

\section{Motivation}
\begin{frame}

  \frametitle{Anwendersicht: Was man gerne hätte ...}
  \begin{itemize}
    \item Automatische Benachrichtigung über Feed-Updates
    \item Aktualität der Feeds
    \item Definition von Filtern auf höherer Ebene: zur \dots
      \begin{list}{}{}
        \item \dots individuellen Vorselektion der News
	\item \dots anbieterübergreifenden Zusammenstellung neuer Feeds
      \end{list}
  \end{itemize}

\end{frame}

\begin{frame}

  \frametitle{Stand der Dinge}
  \begin{itemize}
    \item RSS: Kein Pub/Sub im herkömmlichen Sinne $\rightarrow$ Client/Server
    \item Feeds müssen von Anwendersoftware erfragt werden (Polling)
    \item Vordefinierte Channel $\rightarrow$ thematische Zuordnung auf Publisherseite
    \item Definition von Filtern nur auf Anwenderseite möglich
  \end{itemize}

  \pause
  Probleme:
  \begin{itemize}
    \item Polling:
      \begin{list}{$\longrightarrow$}{}
        \item Großer Datenverkehr im Netz
	\item Hohe serverseitige Last
      \end{list}
    \item Filterdefinition nur beim Anwender:
      \begin{list}{$\longrightarrow$}{}
        \item alle Feeds müssen heruntergeladen werden
      \end{list}
  \end{itemize}
  \nocite{Hicks:2004:RSSBandwith}
  
\end{frame}

\begin{frame}

  \frametitle{Ziel}
  \begin{itemize}
    \item Entlastung der RSS-Server
    \item Reduzierung des Datenverkehrs aufgrund von Polling
    \item Erhöhung des Aktualitätsgrades der Feeds auf Nutzerseite
    \item Ermöglichung der subscriber-seitigen Definition von Filtern
  \end{itemize}

\end{frame}

\section{Umsetzung}
\begin{frame}
  \frametitle{Integration von RSS in verteiltes Pub/Sub-System}
  \begin{itemize}
    \item RSS-Server könnte als Publisher fungieren
    \item Broker sorgen für eine Verteilung der Feeds an Nutzer (Subscriber)
  \end{itemize}

  \pause
  Problem:\\
  Neue Software muss sowohl auf Server-Seite als auch auf Anwenderseite installiert werden.
%    \item Verteilung der RSS-Feeds mittels herkömmlichem Pub/Sub $\Longrightarrow$ Notifikation der Subscriber
\end{frame}

\begin{frame}
  \frametitle{Lösung:}
\begin{itemize}
  \item Rolle der RSS-Server verändert sich nicht
  \item Nutzer holen Feed in der Rolle des Publisher und speisen ihn in das Broker-Netz
  \item Broker verteilen Feeds an Nutzer bzw. Subscriber
\end{itemize}
\end{frame}

\begin{frame}
  \includegraphics[bb=-100 0 200 250,scale=0.6]{RSSPubSub}
\end{frame}

\begin{frame}
  \frametitle{Ansatz: Ein Publisher}
  \begin{itemize}
    \item Ein Publisher würde ausreichen um Feed herunterzuladen
    \item Alle weiteren Nutzer  bzw. Subscriber würden den Feed übers Netzwerk erhalten
  \end{itemize}

  \pause
  Vorteil:\\
  Wenige Anfragen an den RSS-Server
\vspace{0.5cm}
  
  \pause
  Problem:\\
  Aktualität der Feeds!\\
  $\longrightarrow$ lange Übertragungszeiten in Abhängigkeit von Netzstruktur
\end{frame}

\begin{frame}
  \frametitle{Aktualität vs. Server-Belastung}
  \includegraphics[bb=20 0 200 250,scale=0.5]{Verzoegerung}
\end{frame}

\begin{frame}
  \frametitle{Ansatz: Mehrere Publisher}
  \begin{itemize}
    \item Mehrere Publisher laden Feed herunter
    \item $\rightarrow$ Mindestens so viele, dass gewisse Aktualität d. F.'s erreicht wird
    \item $\rightarrow$ Höchstens so viele, dass Server nicht überlastet wird
  \end{itemize}
\vspace{0.5cm}

$\longrightarrow$ Optimum muss bestimmt werden

\end{frame}
